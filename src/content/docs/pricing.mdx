  <div id="calendarOutput"></div>

  <script>
    document.getElementById("calendarForm").addEventListener("submit", function(e){
      e.preventDefault();
      
      const name = document.getElementById("calendarName").value;
      const days = parseInt(document.getElementById("days").value);
      const platform = document.getElementById("platform").value;
      const contentType = document.getElementById("contentType").value;
      const industry = document.getElementById("industry").value;

      let table = `<h2>${name} - Content Calendar</h2>`;
      table += `<table><tr><th>Date</th><th>Time</th><th>Content</th><th>Content Type</th><th>Platform</th><th>Industry</th></tr>`;

      const today = new Date();
      for(let i=0; i<days; i++){
        let date = new Date();
        date.setDate(today.getDate() + i);
        let dateStr = date.toISOString().split('T')[0];

        table += `<tr>
          <td>${dateStr}</td>
          <td>--:--</td>
          <td>Content Idea</td>
          <td>${contentType}</td>
          <td>${platform}</td>
          <td>${industry}</td>
        </tr>`;
      }
      table += `</table>`;
      table += `<button onclick="downloadCSV()">Download as CSV</button>`;

      document.getElementById("calendarOutput").innerHTML = table;
    });

    function downloadCSV(){
      let rows = [];
      document.querySelectorAll("table tr").forEach(tr => {
        let row = [];
        tr.querySelectorAll("th, td").forEach(cell => row.push(cell.innerText));
        rows.push(row.join(","));
      });

      let csvContent = rows.join("\n");
      let blob = new Blob([csvContent], { type: "text/csv" });
      let link = document.createElement("a");
      link.href = URL.createObjectURL(blob);
      link.download = "content_calendar.csv";
      link.click();
    }
  </script>
