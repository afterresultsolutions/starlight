  <script>
    // âœ… Toggle multiple pill selection
    document.querySelectorAll(".cc-pill").forEach(pill => {
      pill.addEventListener("click", () => {
        pill.classList.toggle("active");
      });
    });

    // âœ… Suggested peak posting times (example averages)
    const peakTimes = {
      "Instagram": "11:00",
      "Facebook": "13:00",
      "LinkedIn": "09:00",
      "Twitter": "12:00"
    };

    // âœ… Festival list (example, you can extend this)
    const festivals = [
      { date: "2025-10-02", name: "Gandhi Jayanti" },
      { date: "2025-10-20", name: "Diwali" },
      { date: "2025-12-25", name: "Christmas" }
    ];

    // âœ… Content types when "All" is selected
    const allContentTypes = ["Reels", "Posts", "Carousels", "Stories", "Creatives"];

    function generateCalendar(mode){
      const name = document.getElementById("calendarName").value;
      const startDate = new Date(document.getElementById("startDate").value);
      const endDate = new Date(document.getElementById("endDate").value);
      const platforms = Array.from(document.querySelectorAll(".cc-pill.active")).map(p => p.dataset.value);
      const contentType = document.getElementById("contentType").value;
      const industry = document.getElementById("industry").value;
      const includeFestivals = document.getElementById("includeFestivals").checked;

      if(platforms.length === 0){
        alert("Please select at least one platform.");
        return;
      }
      if(!startDate || !endDate || startDate > endDate){
        alert("Please select a valid date range.");
        return;
      }

      let rows = [];
      rows.push(["Date","Time","Content","Content Type","Platform","Industry"]);

      let dayIndex = 0;
      for(let d = new Date(startDate); d <= endDate; d.setDate(d.getDate()+1)){
        let dateStr = d.toISOString().split('T')[0];
        let todaysTypes = contentType === "All" ? [allContentTypes[dayIndex % allContentTypes.length]] : [contentType];
        
        platforms.forEach(platform => {
          todaysTypes.forEach(type => {
            rows.push([
              dateStr,
              peakTimes[platform] || "--:--",
              "Content Idea",
              type,
              platform,
              industry
            ]);
          });
        });

        dayIndex++;
      }

      // âœ… Add festivals if checked
      if(includeFestivals){
        festivals.forEach(fest => {
          let fDate = new Date(fest.date);
          if(fDate >= startDate && fDate <= endDate){
            rows.push([fest.date, "--:--", fest.name, "Festival", "-", "-"]);
          }
        });
      }

      if(mode === "preview"){
        let table = `<h3>${name} - Content Calendar</h3><table class="cc-table"><tr>${rows[0].map(h=>`<th>${h}</th>`).join("")}</tr>`;
        rows.slice(1).forEach(r=>{
          const isFestival = r[3] === "Festival";
          table += `<tr class="${isFestival ? 'festival':''}">${r.map(c => `<td>${c}</td>`).join("")}</tr>`;
        });
        table += `</table>`;
        document.getElementById("calendarOutput").innerHTML = table;
      }

      if(mode === "download"){
        let csvContent = rows.map(r=>r.join(",")).join("\n");
        let blob = new Blob([csvContent], { type: "text/csv" });
        let link = document.createElement("a");
        link.href = URL.createObjectURL(blob);
        link.download = "content_calendar.csv";
        link.click();
      }
    }
  </script>
