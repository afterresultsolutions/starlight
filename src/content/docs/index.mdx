<script>
  function animateValue(id, start, end, duration) {
    let current = start;
    const obj = document.getElementById(id);
    const range = end - start;
    const increment = end > start ? 1 : -1;
    const stepTime = Math.abs(Math.floor(duration / range));

    const timer = setInterval(function () {
      current += increment;
      obj.innerText = current;
      if (current == end) {
        clearInterval(timer);
      }
    }, stepTime);
  }

  // Run animation after window loads
  window.onload = function () {
    animateValue("projectCount", 0, 1000, 1500);
    animateValue("clientCount", 0, 850, 1500);
    animateValue("campaignCount", 0, 500, 1500);
    animateValue("industryCount", 0, 35, 1500);
  };
</script>
